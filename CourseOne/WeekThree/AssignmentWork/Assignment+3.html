import pandas as pd
from functools import reduce
import re

def populate_energy():
    energy = pd.read_excel('Energy Indicators.xls', skiprows=17, names=['Country', 'Energy Supply', 'Energy Supply per Capita', '% Renewable'], skip_footer=38, na_values='...', usecols=[2,3,4,5])
    energy['Energy Supply'] *= 1000000
        
    energy['Country'] = energy['Country'].map(lambda country: re.sub('\(.*\)|[0-9]', '', country).strip())
    energy.at[energy.index[energy['Country'] == 'Republic of Korea'], 'Country'] = 'South Korea'
    energy.at[energy.index[energy['Country'] == 'United States of America'], 'Country'] = 'United States'
    energy.at[energy.index[energy['Country'] == 'United Kingdom of Great Britain and Northern Ireland'], 'Country'] = 'United Kingdom'
    energy.at[energy.index[energy['Country'] == 'China, Hong Kong Special Administrative Region'], 'Country'] = 'Hong Kong'
    
    return energy

def populate_gdp():
    GDP = pd.read_csv('world_bank.csv', skiprows=4)
    GDP.rename(columns={'Country Name': 'Country'}, inplace=True)
    GDP.at[GDP.index[GDP['Country'] == 'Korea, Rep.'], 'Country'] = 'South Korea'
    GDP.at[GDP.index[GDP['Country'] == 'Iran, Islamic Rep.'], 'Country'] = 'Iran'
    GDP.at[GDP.index[GDP['Country'] == 'Hong Kong SAR, China'], 'Country'] = 'Hong Kong'
    return GDP.filter(regex='(^20(0[6-9]|1[0-5]))|Country$', axis=1)
    
def populate_scimen():
    return pd.read_excel('scimagojr-3.xlsx')
    
def merge_frames(dfs, how):
    merged_df = reduce(lambda left, right: pd.merge(left, right, on='Country', how=how), dfs)
    
    merged_df = merged_df.set_index('Country')
    merged_df = merged_df[['Rank', 'Documents', 'Citable documents', 'Citations', 'Self-citations', 'Citations per document', 'H index', 'Energy Supply', 'Energy Supply per Capita', '% Renewable', '2006', '2007', '2008', '2009', '2010', '2011', '2012', '2013', '2014', '2015']]
    return merged_df

def answer_one():
    energy = populate_energy()
    GDP_last_ten_years = populate_gdp()
    ScimEn = populate_scimen()

    merged_df = merge_frames([energy, GDP_last_ten_years, ScimEn], 'inner')
    top_fifteen = merged_df.loc[merged_df['Rank'] <= 15]
    return top_fifteen
